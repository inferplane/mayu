add a provider · config block 
// providers: one block per upstream; secrets stay as env/file refs.
"providers": {
  "anthropic-direct": { "type": "anthropic",
    "base_url": "https://api.anthropic.com",
    "api_key_ref": { "env": "ANTHROPIC_API_KEY" } },
  "vllm-internal":    { "type": "openai_compatible",
    "base_url": "http://vllm.internal:8000/v1",
    "api_key_ref": { "file": "/secrets/vllm-key" } },
  "bedrock-us":       { "type": "bedrock", "region": "us-west-2",
    "auth": { "mode": "irsa" } }
},
"models": {
  "claude-sonnet-4-6": { "targets": [
    { "provider": "anthropic-direct", "model": "claude-sonnet-4-6" },
    { "provider": "bedrock-us", "model": "anthropic.claude-sonnet-4-6-v1:0", "api": "invoke_model" }
  ] }
}
